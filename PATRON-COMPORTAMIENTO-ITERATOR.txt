function* iterador(col) {
    var nextIndex = 0;
 
    while (nextIndex < col.length) {
        yield col[nextIndex++];//se usa para pausar y reanudar una funciÃ³n generadora
    }
}
 
const x = [1, 2, 3]
const gen = iterador(x)
console.log(gen.next())
// {value: 1, done: false}
console.log(gen.next())
// {value: 2, done: false}
console.log(gen.next())
// {value: 3, done: false}
console.log(gen.next())
// {value: undefined, done: true}
